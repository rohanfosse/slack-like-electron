### 1. Module permissions (`src/renderer/src/utils/permissions.ts`)

```ts
export const ROLE_LEVELS = { student: 0, ta: 1, teacher: 2, admin: 3 } as const
export type Role = keyof typeof ROLE_LEVELS

export function hasRole(userRole: Role, requiredRole: Role): boolean {
  return (ROLE_LEVELS[userRole] ?? -1) >= (ROLE_LEVELS[requiredRole] ?? Infinity)
}

export const ADMIN_MODULES = { /* miroir du backend */ } as const

export function canAccessModule(userRole: Role, module: string): boolean {
  const required = ADMIN_MODULES[module as keyof typeof ADMIN_MODULES]
  return required ? hasRole(userRole, required) : false
}
```
